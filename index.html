function standardizeArgument() { const argument = document.getElementById("argumentInput").value.trim(); if (!argument) { alert("Please enter an argument."); return; } // Expanded list of conclusion indicators const conclusionIndicators = [ "therefore", "thus", "hence", "so", "consequently", "it follows that", "we can conclude that", "which means that", "this implies that", "ergo", "accordingly", "as a result" ]; // Split sentences properly (supports commas, semicolons, periods) let sentences = argument.split(/(?<=[.?!;])\s+/); let premises = []; let conclusion = ""; for (let sentence of sentences) { let lowerSentence = sentence.toLowerCase().trim(); if (conclusionIndicators.some(word => lowerSentence.includes(word))) { conclusion = sentence; // Store detected conclusion } else { premises.push(sentence); } } // Ensure all premises are correctly captured if (premises.length === 0 && conclusion) { premises.push(conclusion); conclusion = ""; } // If no conclusion is explicitly detected, allow the last premise to be the conclusion if (!conclusion && premises.length > 1) { conclusion = premises.pop(); } // Format the output let output = "Standard Form:

"; premises.forEach((premise, index) => { output += `(P${index + 1}) ${premise}
`; }); if (conclusion) { output += `
(C) ${conclusion}`; } else { output += "
No conclusion detected."; } document.getElementById("response").innerHTML = output; }
