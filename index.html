function analyzeArgument() { let text = document.getElementById("argumentInput").value; let premises = []; let conclusion = ""; // Keywords for detecting conclusions const conclusionKeywords = ["therefore", "hence", "thus", "so", "consequently"]; // Smart splitting: Keep "If...then" statements together let sentences = text.match(/[^.!?]+[.!?]+|.+$/g) || []; // This correctly keeps complex statements together for (let sentence of sentences) { let lowerSentence = sentence.toLowerCase().trim(); // If the sentence contains a conclusion indicator, mark it as a conclusion if (conclusionKeywords.some(word => lowerSentence.includes(word))) { conclusion = sentence; } else { premises.push(sentence); // Otherwise, it's a premise } } // Ensure the last sentence is the conclusion if not explicitly found if (premises.length >= 2 && !conclusion) { conclusion = premises.pop(); } // Step 1: Display Standard Form let outputHTML = "

Standard Form:

"; premises.forEach((p, i) => outputHTML += `

P${i+1}: ${p}

`); outputHTML += conclusion ? `

Conclusion: ${conclusion}

` : `

Conclusion: Not explicitly found.

`; // Step 2: Evaluate Argument Validity let isValid = checkValidity(premises, conclusion); outputHTML += isValid ? `

✔ This argument is logically valid.

` : `

✖ This argument is invalid.

`; // Step 3: Detect Fallacies let fallacies = detectFallacies(premises, conclusion); if (fallacies.length > 0) { outputHTML += "

Potential Fallacies Detected:

"; fallacies.forEach(f => outputHTML += `

⚠️ ${f}

`); } else { outputHTML += "

✅ No major logical fallacies detected.

"; } // Step 4: Check Soundness let isSound = checkSoundness(premises); outputHTML += isSound ? `

✔ This argument is sound.

` : `

❌ This argument is unsound.

`; document.getElementById("output").innerHTML = outputHTML; }
