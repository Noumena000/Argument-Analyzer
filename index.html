function checkValidity(standardForm) { const lowerForm = standardForm.toLowerCase(); // Detect Modus Ponens (MP) if (lowerForm.includes("if") && lowerForm.includes("then") && lowerForm.includes("therefore")) { const premise1 = standardForm.match(/p1: if (.*?), then (.*)/i); const premise2 = standardForm.match(/p2: (.*?)/i); const conclusion = standardForm.match(/conclusion: (.*?)/i); if (premise1 && premise2 && conclusion) { const ifClause = premise1[1].trim(); const thenClause = premise1[2].trim(); if (premise2[1].trim() === ifClause && conclusion[1].trim() === thenClause) { return "✓ This argument follows Modus Ponens and is valid."; } } } // Detect Modus Tollens (MT) if (lowerForm.includes("if") && lowerForm.includes("then") && lowerForm.includes("not") && lowerForm.includes("therefore")) { return "✓ This argument follows Modus Tollens and is valid."; } // Detect Hypothetical Syllogism (HS) if (lowerForm.includes("if") && lowerForm.includes("then") && lowerForm.match(/if (.*?), then (.*)/g).length === 2) { return "✓ This argument follows Hypothetical Syllogism and is valid."; } // Detect Disjunctive Syllogism (DS) if (lowerForm.includes("either") && lowerForm.includes("not") && lowerForm.includes("therefore")) { return "✓ This argument follows Disjunctive Syllogism and is valid."; } // Detect Constructive Dilemma if (lowerForm.includes("either") && lowerForm.includes("if") && lowerForm.includes("then") && lowerForm.includes("therefore")) { return "✓ This argument follows Constructive Dilemma and is valid."; } return "X This argument is invalid. The conclusion does not logically follow from the premises."; }
